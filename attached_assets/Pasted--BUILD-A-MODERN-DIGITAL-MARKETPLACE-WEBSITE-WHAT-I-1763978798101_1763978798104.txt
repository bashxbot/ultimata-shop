## WHAT IT DOES:
Sell digital accounts (gaming, streaming, VPN, etc.), combo lists, and offer custom development services (web dev, Python scripts, Telegram bots). Customers buy products, pay online via **PayPal, GCash, or Binance**, and access their purchases in their personal dashboard. NO EMAIL DELIVERY - everything accessed through user panel.

## DATABASE (Prisma Models):

```prisma
model User {
  id            String    @id @default(cuid())
  name          String?
  email         String    @unique
  password      String
  role          Role      @default(USER)
  currency      String    @default("USD") // Preferred currency
  createdAt     DateTime  @default(now())
  orders        Order[]
  tickets       SupportTicket[]
}

enum Role {
  USER
  ADMIN
}

model Product {
  id            String    @id @default(cuid())
  name          String
  slug          String    @unique
  type          ProductType
  categoryId    String
  category      Category  @relation(fields: [categoryId], references: [id])
  description   String
  price         Float
  currency      String    @default("USD")
  stock         Int
  image         String?
  credentials   String?   // Encrypted JSON: {username, password, extra}
  fileUrl       String?   // For combo lists
  fileName      String?
  fileSize      Int?
  status        ProductStatus @default(ACTIVE)
  featured      Boolean   @default(false)
  createdAt     DateTime  @default(now())
  orderItems    OrderItem[]
}

enum ProductType {
  ACCOUNT
  COMBO
}

enum ProductStatus {
  ACTIVE
  DRAFT
  OUT_OF_STOCK
}

model Order {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  total           Float
  currency        String    // PHP, USD
  status          OrderStatus @default(PENDING)
  paymentMethod   PaymentMethod
  paymentId       String?   // Transaction/Reference ID from payment provider
  paymentDetails  String?   // JSON: full payment response
  createdAt       DateTime  @default(now())
  completedAt     DateTime?
  items           OrderItem[]
  payment         Payment?
}

enum OrderStatus {
  PENDING
  PROCESSING
  COMPLETED
  FAILED
  REFUNDED
}

enum PaymentMethod {
  PAYPAL
  GCASH
  BINANCE
}

model Payment {
  id                String    @id @default(cuid())
  orderId           String    @unique
  order             Order     @relation(fields: [orderId], references: [id])
  method            PaymentMethod
  amount            Float
  currency          String
  transactionId     String    @unique // Provider's transaction ID
  status            PaymentStatus
  payerEmail        String?
  payerName         String?
  providerResponse  String?   // JSON: full response from payment gateway
  refundedAmount    Float?
  refundedAt        DateTime?
  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt
}

enum PaymentStatus {
  PENDING
  COMPLETED
  FAILED
  REFUNDED
  PARTIALLY_REFUNDED
}

model OrderItem {
  id                    String    @id @default(cuid())
  orderId               String
  order                 Order     @relation(fields: [orderId], references: [id])
  productId             String
  product               Product   @relation(fields: [productId], references: [id])
  quantity              Int
  price                 Float
  deliveredCredentials  String?   // JSON: copy of credentials at time of purchase
  deliveredAt           DateTime?
}

model ServiceRequest {
  id            String    @id @default(cuid())
  name          String
  email         String
  serviceType   String
  description   String
  budget        String?
  status        String    @default("pending")
  createdAt     DateTime  @default(now())
}

model SupportTicket {
  id          String    @id @default(cuid())
  userId      String
  user        User      @relation(fields: [userId], references: [id])
  subject     String
  status      String    @default("open")
  priority    String    @default("normal")
  createdAt   DateTime  @default(now())
  messages    TicketMessage[]
}

model TicketMessage {
  id            String    @id @default(cuid())
  ticketId      String
  ticket        SupportTicket @relation(fields: [ticketId], references: [id])
  message       String
  isAdminReply  Boolean   @default(false)
  createdAt     DateTime  @default(now())
}

model Coupon {
  id              String    @id @default(cuid())
  code            String    @unique
  discountType    String    // percentage, fixed
  discountValue   Float
  minPurchase     Float?
  expiryDate      DateTime?
  usageLimit      Int?
  usageCount      Int       @default(0)
  active          Boolean   @default(true)
}

model Category {
  id        String    @id @default(cuid())
  name      String
  slug      String    @unique
  products  Product[]
}

model PaymentConfig {
  id        String    @id @default(cuid())
  provider  String    @unique // paypal, gcash, binance
  isActive  Boolean   @default(false)
  isTestMode Boolean  @default(true)
  config    String    // JSON: encrypted API keys, secrets
  lastTest  DateTime?
  updatedAt DateTime  @updatedAt
}

model WebhookLog {
  id          String    @id @default(cuid())
  provider    String    // paymongo, binance, paypal
  event       String
  payload     String    // JSON
  processed   Boolean   @default(false)
  error       String?
  createdAt   DateTime  @default(now())
}
```
